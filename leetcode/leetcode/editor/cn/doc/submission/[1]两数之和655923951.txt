class Solution:
    def twoSum(self, nums: List[int], target: int) -> List[int]:
        map = {}
        for index , value in enumerate(nums):
            map[value] = index
        for j in range(0 , len(nums) , 1):
            if ((target - nums[j]) in map) and (map[target-nums[j]] != j) :
                return [j , map[target-nums[j]]]
# runtime:4 ms
# memory:18.8 MB
